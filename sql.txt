create table empresa(
    id serial primary key,
    nome varchar (50),
    email varchar (50),
    descricao varchar (100),
    ramo varchar (30),
    conta varchar(30),
    senha varchar(64) not null,
    agencia varchar(30),
    banco varchar(50)
);

create table servico(
    id serial primary key,
    id_empresa integer not null,
    descricao varchar(100),
    nome varchar (50),
    preco numeric(5, 2),
    constraint fk_servico_empresa foreign key (id_empresa) references empresa(id)
);

create table cliente(
    id serial primary key,
    nome varchar(50),
    telefone varchar(30),
    email varchar(50)

);

create table servicocliente(
    id serial primary key,
    id_cliente integer not null,
    id_empresa integer not null,
    constraint fk_fita_cliente foreign key (id_cliente) references cliente(id),
    constraint fk_fita_empresa foreign key (id_empresa) references empresa(id)

)
